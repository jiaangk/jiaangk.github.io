## 题目

给出一棵有$N$个节点的树。定义操作为选择abc三个节点满足a与b相连，b与c相连。将与a相连的节点（除了b）的与a的连边和ab之间的连边断掉，之后将a和a相连的节点（除了b）都与c相连。问至少需要多少次操作可以让图变成一张菊花图。

## 题解

将图进行黑白染色，可以看出，本质上这个操作是将a节点颠倒黑白。所以只要对这张图进行染色，之后直接输出较少的颜色的点数$-1$就可以了。

```cpp
#include <bits/stdc++.h>
#define LL long long
#define pb push_back
#define st first
#define nd second
#define INF 0x3f3f3f3f
#define LINF 0x3f3f3f3f3f3f3f3f
template <class T> T read(T &a) {
    a=0;char x=getchar();bool f=0;
    for(;x<'0'||x>'9';x=getchar())f|=x=='-';
	for(;x>='0'&&x<='9';x=getchar())a=(a<<3)+(a<<1)+x-'0';
    if(f)a=-a;
    return a;
}
using namespace std;

const int N = 2e5 + 5;

int cnt[2];
vector<int> G[N];
int c[N];
bool vis[N];

void dfs(int a) {
	vis[a] = 1;
	cnt[c[a]]++;
	int lon = G[a].size();
	for (int i = 0; i < lon; ++i) {
		int to = G[a][i];
		if (!vis[to]) {
			c[to] = 1 ^ c[a];
			dfs(to);
		}
	}
}

int main() {
	int n;
	read(n);
	for (int i = 1; i < n; ++i) {
		int a, b;
		read(a), read(b);
		G[a].pb(b);
		G[b].pb(a);
	}
	dfs(1);
	printf("%d\n", min(cnt[0], cnt[1]) - 1);
    return 0;
}
```
