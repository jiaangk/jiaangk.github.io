## 题意

给出$n$个数，问这些数字有多少种选法，使得选出的数相乘的结果为完全平方数（不能一个都不选）。数据保证所有数的质因子都小于等于$500$且方案总数小于等于$2^{63} - 1$。

## 题解

一摸到题目一筹莫展，线性代数果然还是一窍不通啊！

分析题目，容易得到相乘的结果中所得的数的每个质因子出现次数必须为偶数才能使结果为完全平方数。考虑用唯一分解式来表示每个数。使用例子中的$\{6, 10, 15\}$，只有$\{2, 3, 5\}$是这些数的质因数。所以可以用三个方程来表示选择的数的相应质因数出现次数是不是偶数。$(x_1 + x_2) \% 2 == 0$这是对于$2$来说的，因为$6$和$10$中均出现了奇数次数的$2$，同理$(x_1 + x_3) \% 2 == 0$，$(x_2 + x_3) \% 2 == 0$。由于只与奇偶数有关，我们可以列出线性异或方程组。

$ \begin{bmatrix} 1 & 1 & 0 \\ 1 & 0 & 1 \\ 0 & 1 & 1 \end{bmatrix} $这个方程组有一个自由变量，所以答案为$2 ^ 1 - 1$。$-1$的原因是因为不能全部都不选。

所以答案就是$2 ^ x - 1$，$x$表示自由变量数。

```cpp
#include <bits/stdc++.h>
#define LL long long
#define pb push_back
#define st first
#define nd second
#define INF 0x3f3f3f3f
#define LINF 0x3f3f3f3f3f3f3f3f
template <class T> T read(T &a) {
    a=0;char x=getchar();bool f=0;
    for(;x<'0'||x>'9';x=getchar())f|=x=='-';
	for(;x>='0'&&x<='9';x=getchar())a=(a<<3)+(a<<1)+x-'0';
    if(f)a=-a;
    return a;
}
using namespace std;

int prim[599];
int cnt = 0;
bool vis[600];

void init() {
    for (int i = 2; i <= 500; ++i) {
        if (!vis[i]) {
            prim[++cnt] = i;
        }
        for (int j = 1; j <= cnt && prim[j] * i <= 500; ++j) {
            vis[prim[j] * i] = 1;
            if (i % prim[j] == 0) break;
        }
    }
}

bitset<105> s[100];

void fun(LL a) {
    for (int i = 1; i <= cnt; ++i) {
        vis[i] = 0;
    }
    for (int i = 1; i <= cnt; ++i) {
        while (a % prim[i] == 0) {
            a /= prim[i];
            vis[i] ^= 1;
        }
    }
}

int main() {
    init();
    int T;
    read(T);
    while (T--) {
        int n;
        memset(s, 0, sizeof s);
        read(n);
        for (int i = 1; i <= n; ++i) {
            LL a;
            read(a);
            fun(a);
            for (int j = 1; j <= cnt; ++j) {
                if (vis[j]) s[j][i] = 1;
            }
        }
        int now = 1, tot = 0;
        for (int i = 1; i <= cnt; ++i) {
            while (s[i][now] == 0 && now <= n) {
                for (int j = i + 1; j <= cnt; ++j) {
                    if (s[j][now]) {
                        swap(s[i], s[j]);
                        break;
                    }
                }
                if (!s[i][now]) now++;
            }
            if (now <= n) {
                tot++;
                for (int j = i + 1; j <= cnt; ++j) {
                    if (s[j][now]) s[j] ^= s[i];
                }
            } else break;
        }
        printf("%lld\n", (1ll << n - tot) - 1);
    }
	return 0;
}
```
