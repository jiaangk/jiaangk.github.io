```cpp
struct tp{
	int d, id;
}d[2000];

int ans [2000];

bool cmp(tp a, tp b) {
	if (a.d != b.d) return a.d > b.d;
	return a.id < b.id;
}
class Bonuses
{
        public:
        vector <int> getDivision(vector <int> points)
        {
            int n = points.size();
            int tot = 0;
			for (int i = 0; i < n; ++i) {
            	tot += points[i];
			}
			for (int i = 1; i <= n; ++i) {
				d[i].d = points[i - 1];
				d[i].id = i;
			}
			sort(d + 1, d + n + 1, cmp);
			int totans = 0;
			for (int i = 1; i <= n; ++i) {
				totans += (ans[d[i].id] = 100 * d[i].d / tot);
			}
			for (int i = 1; i <= 100 - totans; ++i) {
				ans[d[i].id]++;
			}
			vector<int> as;
			for (int i = 1; i <= n; ++i) {
				as.pb(ans[i]);
			}
			return as;
        }
};
```

T2代码：

```cpp
struct tp{
	int n, m, tm;
}q[60];

int n, m, tot[60], d[60][60];

bool cmp(tp a, tp b) {
	return a.tm < b.tm;
}

pair<int, int> rotate(int now, int goal) {
	int ans = abs(goal - now);
	return make_pair(min(ans, m - ans), goal);
}

int minl() {
	vector<int> ans;
	ans.clear();
	ans.pb(1);
	for (int i = 2; i <= m; ++i) {
		if (tot[i] > tot[ans[0]]) {
			ans.clear();
			ans.pb(i);
		} else {
			if (tot[i] == tot[ans[0]]) {
				ans.pb(i);
			}
		}
	}
	return ans[0];
}

class VendingMachine
{
    public:
    int motorUse(vector <string> prices, vector <string> purchases)
    {
    	
        n = prices.size(), m = 0;
        memset(d, 0, sizeof d);
        memset(q, 0, sizeof q);
        memset(tot, 0, sizeof tot);
        for (int i = 0; i < n; ++i) {
			int lon = prices[i].size(), nm = 0;
			bool ok = 0;
			for (int j = 0; j < lon; ++j) {
				if (prices[i][j] == ' ') {
					ok = 0;
					continue;
				}
				if (prices[i][j] >= '0') {
					if (!ok) {
						ok = 1;
						if (i == 0) nm = ++m; else nm++;
					}
					d[i + 1][nm] = d[i + 1][nm] * 10 + prices[i][j] - '0';
				}
			}
		}
		int ans = 0, k = purchases.size();
		for (int i = 0; i < k; ++i) {
			int stage = 0;
			for (int j = 0; j < purchases[i].size(); ++j) {
				if (purchases[i][j] == ',' || purchases[i][j] == ':') {
					stage++;
					continue;
				}
				if (stage == 0) {
					q[i + 1].n = q[i + 1].n * 10 + purchases[i][j] - '0';
				}
				if (stage == 1) {
					q[i + 1].m = q[i + 1].m * 10 + purchases[i][j] - '0';
				}
				if (stage == 2) {
					q[i + 1].tm = q[i + 1].tm * 10 + purchases[i][j] - '0';
				}
			}
		}
		for (int i = 1; i <= k; ++i) {
			q[i].n++;
			q[i].m++;
		}
		sort(q + 1, q + k + 1, cmp);
		for (int i = 1; i <= n; ++i) {
			for (int j = 1; j <= m; ++j) {
				tot[j] += d[i][j];
			}
		}
		pair<int, int> bk = rotate(1, minl());
		ans += bk.st;
		int lst = 0;
		for (int i = 1; i <= k; ++i) {
			if (q[i].tm - lst >= 5) {
				bk = rotate(bk.nd, minl());
				ans += bk.st;
			}
			lst = q[i].tm;
			bk = rotate(bk.nd, q[i].m);
			ans += bk.st;
			if (d[q[i].n][q[i].m]) {
				tot[q[i].m] -= d[q[i].n][q[i].m];
				d[q[i].n][q[i].m] = 0;
			} else return -1;
		}
		return ans + rotate(bk.nd, minl()).st;
    }
};
```

T3是一道简单的动态规划，~~但是我被细节卡了好久~~。定义状态$f_{i, j, k, 0/1}$为在距离为$i$的点，高度为$j$，使用过的路标为$k$的情况下，是否到曾经达过最高点，之后进行递推转移即可。

```cpp
LL f[105][52][52][2];
LL ans;
int n, m, w;

class HillHike
{
    public:
    long long numPaths(int distance, int maxHeight, vector <int> landmarks)
    {
        memset(f, 0, sizeof f);
        if (landmarks.size()) {
        	if (landmarks[0] == 0) f[0][0][1][0] = 1;
        	else f[0][0][0][0] = 1;
		} else f[0][0][0][0] = 1;
        n = distance, m = maxHeight;
        w = landmarks.size();
        for (int i = 0; i < n; ++i) {
        	for (int j = 0; j <= m; ++j) {
        		for (int k = 0; k <= w; ++k) {
        			for (int b = 0; b < 2; ++b) {
						if (j < m) {
							if (k < w) {
								if (j + 1 == landmarks[k]) {
									f[i + 1][j + 1][k + 1][b == 0? (j + 1 == m): b] += f[i][j][k][b];
								} else {
									f[i + 1][j + 1][k][b == 0? (j + 1 == m): b] += f[i][j][k][b];
								}
							} else f[i + 1][j + 1][k][b == 0? (j + 1 == m): b] += f[i][j][k][b];
						}
						if (j > 1 - (i == n - 1)) {
							if (k < w) {
								if (j - 1 == landmarks[k]) {
									f[i + 1][j - 1][k + 1][b] += f[i][j][k][b];
								} else {
									f[i + 1][j - 1][k][b] += f[i][j][k][b];
								}
							} else f[i + 1][j - 1][k][b] += f[i][j][k][b];
						}
						if (j > 0) {
							if (k < w) {
								if (j == landmarks[k]) {
									f[i + 1][j][k + 1][b] += f[i][j][k][b];
								} else {
									f[i + 1][j][k][b] += f[i][j][k][b];
								}
							} else f[i + 1][j][k][b] += f[i][j][k][b];
						}
					}
				}
			}
		}
        return f[n][0][w][1];
    }
};
```
